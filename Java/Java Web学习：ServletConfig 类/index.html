    еҪ“ servlet зЁӢеәҸй…ҚзҪ®дәҶеҲқе§ӢеҢ–еҸӮж•°еҗҺпјҢweb е®№еҷЁеңЁеҲӣе»ә servlet е®һдҫӢеҜ№иұЎж—¶пјҢдјҡиҮӘеҠЁе°ҶиҝҷдәӣеҲқе§ӢеҢ–еҸӮж•°е°ҒиЈ…еҲ° ServletConfig еҜ№иұЎдёӯгҖӮеңЁи°ғз”Ё servlet зЁӢеәҸзҡ„ init ж–№жі•ж—¶пјҢжҲ‘д»¬еҸҜд»ҘиҺ·еҸ–еҲ° ServletConfig еҜ№иұЎпјҢжӯӨж—¶е°ұеҸҜд»Ҙеҫ—еҲ°еҪ“еүҚ servlet зҡ„еҲқе§ӢеҢ–еҸӮж•°дҝЎжҒҜгҖӮ

    дёүгҖҒиҺ·еҸ–еҲқе§ӢеҸӮж•°еҖј

    ServletConfig е®һдҫӢдё»иҰҒжңүдёүеӨ§еҠҹиғҪпјҡ

    • иҺ·еҸ– servlet зЁӢеәҸеҲ«еҗҚпјӣ
    • иҺ·еҸ– servlet зҡ„еҲқе§ӢеҢ–еҸӮж•°пјӣ
    • иҺ·еҸ– ServletContext дёҠдёӢж–ҮеҜ№иұЎгҖӮ

    @Override
public void init(ServletConfig servletConfig) throws ServletException {
    System.out.println("init зЁӢеәҸиў«жү§иЎҢ");

    // 1. иҺ·еҸ– servlet зЁӢеәҸзҡ„еҲ«еҗҚ
    System.out.println(servletConfig.getServletName());

    // 2. иҺ·еҸ– init-param зҡ„еҸӮж•°
    // иҺ·еҸ–жүҖжңүеҸӮж•°еҗҚ
    System.out.println("жүҖжңүеҸӮж•°еҗҚ: " + servletConfig.getInitParameterNames());
    // иҺ·еҸ–жҢҮе®ҡеҸӮж•°зҡ„еҸӮж•°еҖј
    System.out.println("username: " + servletConfig.getInitParameter("username"));

    // 3. иҺ·еҸ– ServletContext
    System.out.println(servletConfig.getServletContext());
}
