    еүҚиЁҖ

    PostgreSQL ж•°жҚ®еә“ж”ҜжҢҒеӨҡз§ҚеӨҚеҲ¶и§ЈеҶіж–№жЎҲпјҢд»Ҙжһ„е»әй«ҳеҸҜз”ЁжҖ§пјҢеҸҜдјёзј©пјҢе®№й”ҷзҡ„еә”з”ЁзЁӢеәҸпјҢе…¶дёӯд№ӢдёҖжҳҜйў„еҶҷж—Ҙеҝ—пјҲWALпјүдј йҖҒгҖӮиҜҘи§ЈеҶіж–№жЎҲе…Ғи®ёдҪҝз”ЁеҹәдәҺж–Үд»¶зҡ„ж—Ҙеҝ—дј йҖҒжҲ–жөҒеӨҚеҲ¶пјҢжҲ–иҖ…еңЁеҸҜиғҪзҡ„жғ…еҶөдёӢпјҢе°ҶдёӨз§Қж–№жі•з»“еҗҲдҪҝз”ЁжқҘе®һзҺ°еӨҮз”ЁжңҚеҠЎеҷЁгҖӮ

    й»ҳи®Өжғ…еҶөдёӢпјҢжөҒеӨҚеҲ¶жҳҜејӮжӯҘзҡ„пјҢе…¶дёӯеңЁе°ҶдәӢеҠЎжҸҗдәӨеҲ°дё»жңҚеҠЎеҷЁеҗҺе°Ҷж•°жҚ®еҶҷе…ҘеӨҮз”ЁжңҚеҠЎеҷЁгҖӮиҝҷж„Ҹе‘ізқҖеңЁдё»жңҚеҠЎеҷЁдёӯжҸҗдәӨдәӢеҠЎдёҺжӣҙж”№еңЁеӨҮз”ЁжңҚеҠЎеҷЁдёӯеҸҳеҫ—еҸҜи§Ғд№Ӣй—ҙеӯҳеңЁеҫҲе°Ҹзҡ„е»¶иҝҹгҖӮиҝҷз§Қж–№жі•зҡ„дёҖдёӘзјәзӮ№жҳҜпјҢеҰӮжһңдё»жңҚеҠЎеҷЁеҙ©жәғпјҢеҲҷеҸҜиғҪж— жі•еӨҚеҲ¶д»»дҪ•жңӘжҸҗдәӨзҡ„дәӢеҠЎпјҢиҝҷеҸҜиғҪеҜјиҮҙж•°жҚ®дёўеӨұгҖӮ

    жң¬ж¬Ўе®һйӘҢе°ҶиҰҒеңЁ Docker дёҠе®үиЈ… PostgreSQL13 е№¶й…ҚзҪ®дё»д»ҺзҺҜеўғгҖӮдёәдәҶз®ҖеҢ–жј”зӨәзҺҜеўғпјҢиҝҷйҮҢеҸӘз”ЁдёҖеҸ°жңҚеҠЎеҷЁжқҘжј”зӨәпјҢйҖҡиҝҮдёҚеҗҢз«ҜеҸЈжқҘеҢәеҲҶгҖӮ

    # 1. еҲ йҷӨе®№еҷЁ
docker rm -f slave
# 2. еҲ йҷӨеҺҹжңүж–Үд»¶еӨ№пјҢе°Ҷ repl йҮҚе‘ҪеҗҚдёә slave
cd /data/psql/
rm -rf slave
mv repl slave
cd /data/psql/slave
# 3. жҹҘзңӢй…ҚзҪ®дҝЎжҒҜ
# postgresql.auto.conf е°Ҷеҗ«жңүеӨҚеҲ¶жүҖйңҖдҝЎжҒҜ
cat postgresql.auto.conf

primary_conninfo = 'user=repuser password=123456 channel_binding=prefer host=172.18.0.101 port=5432 sslmode=prefer sslcompression=0 ssl_min_protocol_version=TLSv1.2 gssencmode=prefer krbsrvname=postgres target_session_attrs=any'

    йҮҚе»ә slave е®№еҷЁпјҡ

    docker run -d \
--network dockernetwork --ip 172.18.0.102 -p 5433:5432 \
--name slave -h slave \
-e "POSTGRES_DB=postgres" \
-e "POSTGRES_USER=postgres" \
-e "POSTGRES_PASSWORD=postgres" \
-v /data/psql/slave:/var/lib/postgresql/data \
postgres
