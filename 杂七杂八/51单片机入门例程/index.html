uint * split_num(uint n)  // е°ҶеӣӣдҪҚж•°жӢҶеҲҶдёәзӢ¬з«Ӣзҡ„еӣӣдёӘж•°еӯ—
{
    uint nums[4];

    nums[0] = n / 1000;         // еҚғдҪҚ
    nums[1] = n % 1000 / 100;   // зҷҫдҪҚ
    nums[2] = n % 100 / 10;     // еҚҒдҪҚ
    nums[3] = n % 10;           // дёӘдҪҚ

    return nums;
}


void main()
{
    uint i, j, x, L1, *nums;

    while(1)
    {   
        for(i=0 ; i < 10000 ; i++)  // iдёәеңЁж•°з Ғз®ЎдёҠжҳҫзӨәзҡ„4дҪҚж•°
        {
            nums = split_num(i);      // жӢҶеҲҶж•°еӯ—
            for(x=0 ; x<100 ; x++)    // иҜҘеұӮеҫӘзҺҜжҺ§еҲ¶ж•°еӯ—еўһй•ҝйҖҹзҺҮ
            {
                L1 = 0xF7;              // еҲқе§ӢдҪҚйҖү
                for(j=0 ; j<4 ; j++)    // иҜҘеұӮи®©еӣӣдёӘж•°з Ғз®ЎеҫӘзҺҜжҳҫзӨәж•°еӯ—пјҢеҲ©з”ЁдҪҷжҷ–ж•Ҳеә”еҲ¶йҖ еҮәеҗҢж—¶жҳҫзӨәзҡ„ж•Ҳжһң
                {
                    P1 = table[*(nums + j)];  // ж®өйҖү
                    P0 = L1;                                // дҪҚйҖү
                    delay(1);                 // е»¶ж—¶
                    L1 >>= 1;                 // еҸіз§»дёҖдҪҚж•°з Ғз®Ў
                    P0 = 0x00;                // ж¶ҲеҪұ
                }
            }
        }
    }
}
