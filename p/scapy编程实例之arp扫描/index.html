from multiprocessing import Queue, Process

from scapy.layers.l2 import Ether, ARP

def scan(network, func):
    """
    æ‰«æä¸»æœº
    :param network: æ‰«æçš„ç½‘æ®µ
    :param func: æ‰«æè°ƒç”¨çš„å‡½æ•°

    """
    queue = Queue()

        arp = Process(target=func, args=(ip, queue))  # åˆ›å»ºè¿›ç¨‹
        arp.start()  # å¼€å§‹è¿›ç¨‹

    return successful_mac_list
