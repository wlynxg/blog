import abc

class Base(metaclass=abc.ABCMeta):
    @abc.abstractmethod
    def say(self):
        pass

class Dog(Base):
    def say(self):
        print('I am a dog.')

class Cat(Base):
    def say(self):
        print('I am a cat.')


animal = []
dog = Dog()
animal.append(dog)
cat = Cat()
animal.append(cat)
for i in animal:
    i.say()
