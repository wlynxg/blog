class Demo:
    def __iter__(self):
        self.a = 0
        self.b = 1
        return self

    def __next__(self):
        self.a, self.b = self.b, self.a + self.b
        return self.a


num = Demo()

        yield a
        a, b = b, a + b
        counter += 1
