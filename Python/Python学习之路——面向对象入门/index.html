    class Myclass(object):
    # зұ»еұһжҖ§
    name = 'cls'

    def __init__(self, value):
    # е®һдҫӢеұһжҖ§
        self.value = value

    # е®һдҫӢж–№жі•
    def test1(self):
        print(self.value)

    # зұ»ж–№жі•
    @classmethod
    def tset2(cls):
        print(cls.name)

    # йқҷжҖҒж–№жі•
    @staticmethod
    def test3(str1):
        print(str1, 'World!')

a = Myclass('good')
a.name = 'self'

print(a.value) # good
print(a.name, ' || ', Myclass.name) # self  ||  cls

a.test1() # good
Myclass.test1(a) # good

a.tset2() # cls
Myclass.tset2() # cls

a.test3('Hello') # Hello World!
Myclass.test3('Hello') # Hello World!
